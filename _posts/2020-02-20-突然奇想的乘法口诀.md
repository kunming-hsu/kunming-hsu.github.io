###SQL显示九九乘法口诀

突发奇想，兴趣为之！

```sql
DECLARE @n INT=1
DECLARE @m INT=1
DECLARE @sql NVARCHAR(100)
CREATE TABLE #temp
  (
     num INT
  )
WHILE @n <= 9
  BEGIN
      INSERT INTO #temp
                  (num)
      VALUES     (@n)
     
      exec('Alter   table #temp  add '+'c'+@n+'  nvarchar(100)')
      WHILE @m <= @n
        BEGIN
            SET @sql='''' + CONVERT(NVARCHAR(10), @m) + '*'
                     + CONVERT(NVARCHAR(10), @n) + '='
                     + CONVERT(NVARCHAR(10), @n*@m) + ''''
            EXEC('update  #temp   set  '+'c'+@m+'='+@sql+' where' +' num'+'='+@n)
            SET @m=@m + 1
        END
      SET @m=1
      SET @n=@n + 1
  END
SELECT *
FROM   #temp
ORDER  BY num
DROP TABLE #temp
```
